function show_message()
{
    print "I am sample void function";
}

function mul(a: int, b: int) : int
{
    return a*b;
}

function add(a: int, b: int) : int
{
    return a+b;
}

function fact(n: int) : int {
    if (n is 1)
        return 1;
    return n*fact(n-1);
}

function main(argc: int)
{
    new a: int;
    new b: int;
    input a;
    input b;

    new sum: int = add(a, b);
    print sum;

    new n: int;
    input n;
    print fact(n);

    show_message();
}
