class Int {
    public value: int;

    public function Int(val: int) {
        value = val;
    }

    public function get() : int {
        return value;
    }

    public function set(i: int) {
        value = i;
    }

    public function add(i: int) : int {
        value = value + i;
        return value;
    }
};

function foo(i: Int) : int
{
    print i.get();
}

function main()  # entry point
{
    new I1: Int = Int(5);
    new I2: Int = I1;
    foo(I2);
    new i: Int;
}
