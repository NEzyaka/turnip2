class Int {
    public value: int;

    public function Int(val: int) {
        this.value = val;
    }

    public function get() : int {
        return this.value;
    }

    public function set(i: int) {
        this.value = i;
    }

    public function add(i: int) : int {
        this.value = this.value + i;
        return this.value;
    }
};

function foo(i: Int) : int
{
    i.set(7); # correct
}

function test() : Int {
    var I: Int = Int(10);
    I.set(1000);
    return I;
}

function main()  # entry point
{
    var I: Int = Int(5);
    foo(I);
    println I.get();
    println test().value;
}
