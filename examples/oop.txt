class Int {
    public value: int;

    public function Int(val: int) {
        this.value = val;
    }

    public function get() : int {
        return this.value;
    }

    public function set(i: int) {
        this.value = i;
    }

    public function add(i: int) : int {
        this.value = this.value + i;
        return this.value;
    }
};

function foo(i: Int) : int
{
    i.value = 7;
}

function main()  # entry point
{
    new I1: Int = Int(5);
    new I2: Int = I1;
    foo(I2);
    print I2.get();
}
