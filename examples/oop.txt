class Int {
    private value: int;

    public function Int(val: int) {
        this.value = val;
    }

    public function get() : int {
        return this.value;
    }

    public function set(i: int) {
        this.value = i;
    }

    public function add(i: int) : int {
        this.value = this.value + i;
        return this.value;
    }

    public function getptr() : Int {
        return this;
    }
};

function foo(i: Int) : int
{
    i.set(7); # correct
}

function test() : Int {
    new I: Int = Int(10);
    I.set(7);
    return I;
}

function main()  # entry point
{
    new I: Int = Int(5);

    if I.get() is 5 {
        new a: int = 5;
        print a;
    }

    print a;
}
